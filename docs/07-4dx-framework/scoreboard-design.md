```dart
// Progress bar visualization rules
class ProgressBarStyle {
  // Fill color based on achievement
  Color getFillColor(double percentage) {
    if (percentage >= 100) return Colors.green;      // Target met
    if (percentage >= 80) return Colors.lightGreen;  // Almost there
    if (percentage >= 60) return Colors.amber;       // Needs work
    return Colors.red;                                // Critical
  }
  
  // Show "exceeded" glow effect when > 100%
  bool showGlowEffect(double percentage) => percentage > 100;
  
  // Cap display at 150% even if actual is higher
  double displayPercentage(double actual) => min(150, actual);

class ScoreGauge extends StatelessWidget {
  final double score;
  final int rank;
  final int totalMembers;
  
  @override
  Widget build(BuildContext context) {
    return CircularPercentIndicator(
      radius: 80.0,
      lineWidth: 12.0,
      percent: (score / 100).clamp(0.0, 1.5),
      center: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Text(score.toStringAsFixed(1), style: titleLarge),
          Text('SCORE', style: caption),
        ],
      ),
      progressColor: _getScoreColor(score),
      backgroundColor: Colors.grey.shade200,
      circularStrokeCap: CircularStrokeCap.round,
      footer: Text('#$rank of $totalMembers'),
    );
  }
}
```

### 2. Measure Progress Bar

```dart
class MeasureProgressBar extends StatelessWidget {
  final String measureName;
  final int actual;
  final int target;
  final double percentage;
  
  @override
  Widget build(BuildContext context) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Row(
          mainAxisAlignment: MainAxisAlignment.spaceBetween,
          children: [
            Text(measureName),
            Text('$actual/$target  ${percentage.toStringAsFixed(0)}%'),
          ],
        ),
        LinearProgressIndicator(
          value: (percentage / 100).clamp(0.0, 1.5),
          backgroundColor: Colors.grey.shade200,
          valueColor: AlwaysStoppedAnimation(_getColor(percentage)),
        ),
        if (percentage < 100)
          Text('Need ${target - actual} more', style: caption),
        if (percentage >= 100)
          Row(children: [Icon(Icons.star), Text('Target met!')]),
      ],
    );
  }
}
```
